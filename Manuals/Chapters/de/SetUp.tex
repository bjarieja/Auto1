\chapter{Inbetriebnahme}

\section{Schritt 1: Standortwahl und Vorbereitung}

\begin{itemize}
	\item \textbf{Untergrund:} Wählen Sie eine ebene, stabile und trockene Arbeitsfläche (Tisch oder Werkbank).
	
	\item \textbf{Platzbedarf:} Achten Sie darauf, dass an beiden Enden des Transportbands genügend Platz für das Auflegen und Entnehmen der Transportboxen ist.
	
	\item \textbf{Umgebung:} Stellen Sie sicher, dass keine losen Gegenstände oder Kabel in die Nähe der Laufrollen gelangen können.

\end{itemize}

\section{Schritt 2: Mechanischer Aufbau}

\begin{itemize}
	\item \textbf{Ausklappen:} Klappen Sie die Stützbeine nacheinander vollständig aus.
	
	\item \textbf{Ausrichtung:} Kontrollieren Sie ob das Tischförderband gerade steht. Schiefstellung kann dazu führen, dass die Transportboxen zur Seite rutschen oder herunterfallen.
	
	\item \textbf{Mittigkeit:} Prüfen Sie, ob das Transportband mittig auf den Laufrollen liegt.
	
	\item \textbf{Spannung:} Drücken Sie leicht mit einem Finger von unten mittig auf das Transportband. Es sollte stramm sitzen, aber eine minimale Flexibilität von höchstens  0,5-1,0 cm haben.
	
	% \ref{sec:Vorspannung einstellen}  Vorspannung anpassen
	
	\item \textbf{Fremdkörper:} Stellen Sie sicher, dass keine Werkzeuge oder Verpackungsreste auf dem \textbf{Transportband} liegen.
	
\end{itemize}

\section{Schritt 3: Elektrischer Anschluss}

\begin{itemize}
	\item \textbf{Sichtprüfung:} Überprüfen Sie das Gehäuse des \textbf{Bedienelements} und das \textbf{Netzteil} auf sichtbare Schäden.
	
	\item \textbf{Kontrolle Hauptschalter:} Stellen Sie sicher, das der \textbf{Hauptschalter} am \textbf{Bedienelement} in der Postition ''O'' steht. 
	
	\item \textbf{Verbindung:} Stecken Sie den \textbf{DC-Stecker} des \textbf{Netzteils} in die  \textbf{Buchse} am \textbf{Bedienelement} des \textbf{Tischförderbandes}.
	
	\item \textbf{Netzanschluss:} Stecken Sie das \textbf{Netzteil} in eine gut erreichbare Steckdose (100--240 V).
	
\end{itemize}

\section{Schritt 4: Inbetriebnahme}
\label{sec:Inbetriebnahme}

% siehe Flowchart Erstes Einschalten

\begin{itemize}
	\item \textbf{Erstes Einschalten: } Nachdem Sie die Schritte 1 - 3 vollständig durchgeführt haben, kann das \textbf{Tischförderband} nun zum ersten Mal eingeschaltet werden.
	Stellen Sie zum Einschalten den \textbf{Hauptschalter} auf Position ''I'', das System wird gestartet.
	Auf der \textbf{Anzeige} ist der Startbildschirm zu sehen.
	
	\item \textbf{Freigabe der Lichtschranken:} Die \textbf{Anzeige} fragt Sie nach jedem Einschalten, ob beide \textbf{Lichtschranken} freigegeben sind. Stellen Sie erneut sicher, dass keine \textbf{Transportboxen} oder Fremdkörper auf dem \textbf{Transportband} oder zwischen den \textbf{Lichtschranken} liegen.\\
	Bestätigen Sie die Meldung mit dem \textbf{Quittiertaster}. Die \textbf{Anzeige} zeigt nun das Hauptmenü.
	
	\item \textbf{Funktionsprüfung:}
	
	%Braucht weitere Bearbeitung
	
	\begin{itemize}
		\item Läuft das \textbf{Transportband} ruhig und ohne schleifende Geräusche?
		
		\item Bleibt das \textbf{Transportband} in der Mitte oder wandert er zu einer Seite aus? (Falls ja: Justierschrauben an den Laufrollen vorsichtig anpassen).	
		
		\item \textbf{Modus wählen:} Wählen sie für die Funktionsprüfung mit dem Modusschalter den Modus L aus.  
		
		\item \textbf{Geschwindigkeit auf Minimum:} Drehen Sie den Drehregler auf die kleinste Stufe.
		
		\item \textbf{Starten:} Bitte legen sie eine \textbf{Transportbox} auf die rechte Seite des \textbf{Transportbands}, in die \textbf{Lichtschranke 2}. 
		
	\end{itemize}
	
	\item \textbf{Geschwindigkeitstest:} Erhöhen Sie langsam die Geschwindigkeit, um die Stabilität der \textbf{Stützbeine} bei Vibrationen zu testen.
\end{itemize}



\section{Schritt 5: Betrieb}

\begin{itemize}
	\item \textbf{Aufsichtspflicht:} Bleiben Sie während des Betriebs immer in Reichweite des \textbf{Hauptschalters}.
	
	\item \textbf{Funktionen:} Technische Details und Bedienhinweise finden Sie im Kapitel \ref{sec:Funktionen} \textbf{Funktionen} .
	
	\item \textbf{Not-Aus:} Im Falle einer Blockade stellen Sie sofort den \textbf{Hauptschalter} auf ''O''.
	%Greifen Sie niemals bei laufendem Motor in das Transportband.
	
\end{itemize}
