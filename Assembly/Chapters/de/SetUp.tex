\chapter{Aufbau- und Inbetriebnahmeplan}

\section{Schritt 1: Standortwahl und Vorbereitung}

\begin{itemize}
	\item \textbf{Untergrund:} Wählen Sie eine ebene, stabile und trockene Arbeitsfläche (Tisch oder Werkbank).
	
	\item \textbf{Platzbedarf:} Achten Sie darauf, dass an beiden Enden des \textbf{Transportbands} genügend Platz für die \textbf{Kisten}.
	
	\item \textbf{Umgebung:} Stellen Sie sicher, dass keine losen Gegenstände oder Kabel in die Nähe der Laufrollen gelangen können.
\end{itemize}

\section{Schritt 2: Mechanischer Aufbau}

\begin{itemize}
	\item \textbf{Ausklappen:} Klappen Sie die \textbf{Transportband} nacheinander vollständig aus.
	
	\item \textbf{Ausrichtung:} Kontrollieren Sie ob das Band gerade steht. Ein schiefes Band führt dazu, dass das Transportgut zur Seite rutscht.
\end{itemize}

\section{Schritt 3: Elektrischer Anschluss}

\begin{itemize}
	\item \textbf{Sichtprüfung:} Überprüfen Sie das Gehäuse der Steuereinheit und das Netzteil auf sichtbare Schäden.
	
	\item \textbf{Verbindung:} Stecken Sie den DC-Stecker des Netzteils in die Buchse an der Steuereinheit des Transportförderbandes.
	
	\item \textbf{Netzanschluss:} Stecken Sie das Netzteil in eine gut erreichbare Steckdose (100--240 V).
	
	\item \textbf{Hinweis:} Das Gerät sollte sofort einsatzbereit sein, aber noch nicht anlaufen.
\end{itemize}

\section{Schritt 4: Begutachtung des Transportförderbandes (Vor dem ersten Start)}

\begin{itemize}
	\item \textbf{Mittigkeit:} Prüfen Sie, ob das \textbf{Transportband} mittig auf den Rollen liegt.
	
	\item \textbf{Spannung:} Drücken Sie leicht mit dem Finger, von unten, an die Mitte des \textbf{Transportbands}. Es sollte stramm sitzen, aber eine minimale Flexibilität (ca. 0,5--1 cm Spiel) haben.
	
	\item \textbf{Fremdkörper:} Stellen Sie sicher, dass keine Werkzeuge oder Verpackungsreste auf dem \textbf{Transportband} liegen.
\end{itemize}

\section{Schritt 5: Inbetriebnahme und Testlauf}

\begin{itemize}
	\item \textbf{Modus wählen:} Wählen sie für die Funktionsprüfung mit dem Modusschalter den Modus L aus.  
	
	\item \textbf{Geschwindigkeit auf Minimum:} Drehen Sie den Drehregler auf die kleinste Stufe.
	
	\item \textbf{Starten:} Bitte legen sie eine Kiste auf die rechte Seite des \textbf{Transportbands}, in die Lichtschranke. 
	
	\item \textbf{Funktionsprüfung:}
	\begin{itemize}
		\item Läuft das \textbf{Transportband} ruhig und ohne schleifende Geräusche?
		\item Bleibt das \textbf{Transportband} in der Mitte oder wandert er zu einer Seite aus? (Falls ja: Justierschrauben an den Laufrollen vorsichtig anpassen).
	\end{itemize}
	
	\item \textbf{Geschwindigkeitstest:} Erhöhen Sie langsam die Geschwindigkeit, um die Stabilität der \textbf{klappbaren Stützbeine} bei Vibrationen zu testen.
\end{itemize}

\section{Schritt 6: Regelbetrieb}

\begin{itemize}
	\item \textbf{Aufsichtspflicht:} Bleiben Sie während des Betriebs immer in Reichweite der \textbf{Hauptschalter}.
	
	\item \textbf{Modus:} Technische Details und Bedienhinweise finden Sie in dem Kapitel 6.1 \ref{sec:Grundfunktionen}.
	
	\item \textbf{Not-Halt:} Im Falle einer Blockade drücken Sie sofort die Not-Aus-Taster. 
	%Greifen Sie niemals bei laufendem Motor in das Transportband.
\end{itemize}
