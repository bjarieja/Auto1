\Mysection{Einleitung}

\STANDARD{Sensoren}
{ 
	
  \framesubtitle{Einleitung}	
  \textbf{Grundsätzlich:} Sensor (lat: \glqq Fühler" ) dienen zur \textbf{qualitativen} und \textbf{quantitaiven Auffassung} von physikalischen, chemischen, biologischen, klimatischen und medizinischen Größen. \cite{Hering:2023} \\ 
  Jedes Sensorsystem besteht aus folgenden Komponenten: 
  \begin{itemize}
  	\item \textbf{Sensor-Element} z.B. veränderlicher Widerstand
  	\item \textbf{Auswerte-Elektronik} z.B. Mikrokontroller
  \end{itemize}
}

\STANDARD{Optische Näherungssenoren}
{ 
Optische Näherungssensoren nutzen Sensorelemente, welche sowohl aus \textbf{Transmitter} als auch aus \textbf{Receiver} bestehen. Hierbei sind \textbf{Strahlenquellen} beispielsweise der \textbf{Sender} und \textbf{Fotodioden} die \textbf{Empfänger}. 
\begin{itemize}
	
	\item \textbf{Einweg-Lichtschranken:} Sender und Empfänger gegenüberliegend. Hohe Auflösung, bis zu 10m \cite{Hering:2023}
	\item \textbf{Reflex-Lichtschranken:} Sender und Empfänger in einem Gehäuse, \textbf{Reflektor}. Einfache Einstellung, bis zu 5m Abstand \cite{Hering:2023}
\end{itemize}
}

\Mysection{Aktoren}
\STANDARD{Aktoren}
{ 
\textbf{Grundsätzlich:} Aktoren dienen der \textbf{Erzeugung} von \textbf{Bewegungen} oder der Aufbringung von \textbf{Kräften} und \textbf{Momenten}. Hierbei fasst der Begriff alle Ausgabeelemente die sowohl analog (z.B. Stellmotoren)  als auch binär (z.B. Relais ) wirken können zusammen \cite{Roddeck:2023} 
}

\STANDARD{Motoren}
{ 
\textbf{Grundsätzlich:} Durch das \textbf{Bestromen} einer \textbf{Spule} wird gegenüber einem \textbf{bestehenden Magnetfeld (Permanent- oder Elektromagnet)} eine Kraft aufgebracht, welche dann in einem \textbf{Moment} resultiert. \cite{Feynman:2020} 
\begin{itemize}
	\item \textbf{Moment $\propto$ Stromstärke} 
	\item \textbf{Drehzahl $\propto$ Spannung }
\end{itemize}

\STANDARD{Funktionsprinzip einer Motorregelung}
{ 
\textbf{Grundlegend:} Ein Motortreiber kann sowohl als binärer als auch analoger Verstärker betrachtet werden. 
\cite{Roddeck:2023}
Integriert in einer \textbf{Regelstrecke} gilt folgender Ablauf:
 \begin{itemize}
 	\item Sensor nimmt \textbf{IST-Wert} auf.
 	\item  Mikrokontroller vergleicht \textbf{IST} gegen \textbf{SOLL} (\textbf{Rückkopplung})
 	\item Mikrokontroller regelt mittels \textbf{Model} (z.B. \textbf{PID}) zum \textbf{SOLL-Wert} nach. 
 	\item Motortreiber \textbf{interpretiert} Signal von Mikrokontroller und passt Ausgangssignal an.
 \end{itemize} 
 Der Motortreiber ist Teil der Streck eund nicht des Reglers somit ist er nur ein Aktor
}

\STANDARD{Funktionsprinzip einer Motorregelung}
{ 
	\textbf{Grundlegend:} Ein Motortreiber kann als binärer (PWM) oder analoger Verstärker betrachtet werden \cite{Roddeck:2023}. 
	Er ist Teil der \textbf{Strecke}, bestehend aus Motortreiber und Motor.
} 
\STANDARD{Funktionsprinzip einer Motorregelung} 
{
	Der Ablauf einer geschlossenen Regelung ist wie folgt:
	\begin{itemize}
		\item Ein Sensor erfasst den \textbf{Istwert} (z. B. Drehzahl).
		
		\item Der Mikrokontroller vergleicht \textbf{Istwert} und \textbf{Sollwert} und bildet die \textbf{Regeldifferenz}:
		\[
		e(t) = \text{Soll} - \text{Ist}
		\]
		
		\item Ein \textbf{Regelalgorithmus} (z. B. PID-Regler) berechnet daraus die \textbf{Stellgröße} \(u(t)\).
		
		\item Die \textbf{Stellgröße}  wird über den Motortreiber umgesetzt 
		(z. B. als PWM-Signal oder analoge Spannung).
		
		\item Der Motortreiber beeinflusst die \textbf{Strecke} (Motor), wodurch sich der \textbf{Istwert} ändert.
		
		\item Der neue \textbf{Istwert} wird erneut erfasst (\textbf{Rückkopplung}), wodurch ein kontinuierlicher Regelkreis entsteht.
	\end{itemize} 
	\cite{Roddeck:2023}
}





\STANDARD{Standard Regelungsverfahren - PID}{
PID-Regelung (Proportional-Integral-Differanzial-Regelung) ist eine Methodik um die Differenz zwischen Soll- und Ist-Wert zu minimieren.  \cite{Roddeck:2023}
\begin{figure}
	$u(t)=K_P e(t)+K_I \int_0^t e(\tau)d\tau+K_D \frac{de(t)}{dt}$
\end{figure}


\begin{itemize}
	\item Der Proportionalteil reagiert auf Fehler .
	\item Der Integralteil korrigiert dauerhafte Abweichungen. 
	\item Der Differenzialteil reagiert auf die Änderung des Fehlers.
\end{itemize}
